class Solution {
    public int searchInsert(int[] nums, int target) {
        int temp = 0;
        for (int i = 0; i < nums.length; i++)
        {
            if(target > nums[i])
            {
                temp = i+1;
            }
            if (nums[i] == target)
            {
                return i;
            }
        }
        return temp;
    }
}
